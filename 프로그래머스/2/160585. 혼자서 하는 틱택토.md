import java.util.*;

class Solution {
    public int solution(String[] board) {
        Map<Integer, Character> boardMap = new HashMap<>();
        
        for (int i = 0; i < 3; i++) {
            for (int j = 0; j < 3; j++) {
                boardMap.put(i * 3 + j, board[i].charAt(j));
            }
        }
        
        int checkO = 0;
        int checkX = 0;
        
        for (Character value : boardMap.values()) {
            if (value == 'O') {
                checkO++;
            } else if (value == 'X') {
                checkX++;
            }
        }
        
        // 게임 시작 안 한 경우
        if(checkO == 0 && checkX == 0) return 1;
        
        // "O"의 개수가 "X" 개수보다 작으면 유효하지 않음.
       if (checkO < checkX) return 0;
        
        // 게임 종료 여부 확인
        boolean end = isGameEnded(boardMap);
        
        // 게임이 종료되었음에도 계속 진행 중인 경우 유효하지 않음
        if (end && (checkO != checkX + 1)) return 0;
        
        // 게임 규칙이 지켜짐.
        return 1;
    }
    
    /*
        빙고가 나오면 게임 종료
    */
    private boolean isGameEnded(Map<Integer, Character> boardMap) {

        for(int i = 0; i <= 6; i+=3){
            // 가로 빙고
            if (boardMap.get(i) != '.' && boardMap.get(i) == boardMap.get(i+1) && boardMap.get(i) == boardMap.get(i+2)) return true; 
            // 세로 빙고
            if (boardMap.get(i) == boardMap.get(i+3) && boardMap.get(i) == boardMap.get(i+6)) return true; 
        }
        
        // 대각선 빙고
        if(boardMap.get(0) != '.' && boardMap.get(0) == boardMap.get(4) && boardMap.get(0) == boardMap.get(8)) return true; 
        if(boardMap.get(2) != '.' && boardMap.get(2) == boardMap.get(4) && boardMap.get(2) == boardMap.get(6)) return true; 
        
        return false; // 모든 경우를 확인했지만 승자가 없으면 게임 종료되지 않음
     
    }

}
